-- High Distinction Autumn 2020-- Lhogeshwaran Purushothaman-- lhogeshwaran.purushothaman@student.uts.edu.au-- 13313491-- Product Imaging Reports Database-- (E-commerce platform)-- This database is intended to serve the purposes of reporting for an image processing team-- in an e-commerce company. ---- The unique products in the 'products' table identified by a 'product_id', each belong to a -- category described by 'product_line' (product_line = 'beauty').---- These products can be photographed in a photo studio desctibed by a unique 'studio_code' -- (studio_code = 'FRA') in the 'studio' table and each studio is present in a specific -- region described by 'marketplace' (marketplace = 'DE').---- One product may be photographed in multiple studios and studios may photograph multiple products.---- A single product can have many images each represented by 'image_id' in the 'image_asset' table-- and each image asset has a 'variant' (variant = 'FRNT') naming that specifies the view of the product-- (ex: FRNT to show front view of the product). There is a 'process_class' (process_class = 111) column, -- which is used by the image processing team for their internal reports. ---- Every image identified by 'image_id' is unique and can represent "one and only one product" and be sourced-- from "one and only one studio". However, there can be "many images per product, and as mentioned earlier, -- one product can be photographed by many studios".---- Each 'image_id' can be go through one or multiple image processing steps (retouch, qa, colorauthor)-- defined in the 'workflow_event' (workflow_event = retouch_completed, qa_completed, etc.) of the 'n_points'-- table (Normalized points). However, it is important to note that an image can go through one image -- processing step only once. -- Each image processing step (or worflow_event) has a specific number of points defined to it for quantifying-- the effort taken to process the photo. This is specified in the 'points' column of the 'points_table'.-- The date of the activity is recorded as the 'event_date'.---- There is a team of 1 manager, 3 leads and 12 image processing associates described in the 'team' table.-- The associates (user_id = 'npsanu') report (report_to = 'chandmo') to a specific 'lead' and the leads -- in turn report to a 'manager' (report_to = 'sraju'). No associate or lead be part of more than one team-- or report to more than one lead. The manager is the head the of the team and reports to the director, -- hence does not have a 'reports_to' value in the 'team' table.-- Sometimes the associates help out with the tasks of other teams. For example, an associate of retouch team-- may help with qa or colorauthor activites, if required. Depending on the activity they perform on a photo-- the points will be calculated.-- Sometimes the associates can take a leave of absence recorded by 'is_active' (is_active = FALSE).-- Drop tables if already existing.DROP TABLE products 		CASCADE;DROP TABLE studio 			CASCADE;DROP TABLE image_asset 		CASCADE;DROP TABLE team 			CASCADE;DROP TABLE points_table 	CASCADE;DROP TABLE A simple query of a single table.-- Get the details of image_if, product_id and studio_code for 20 rows and order them in -- descending order by product_id.SELECT image_id,	product_id,	studio_codeFROM image_assetORDER BY product_id DESCLIMIT 20;-- 2. A query which uses the words "natural join".SELECT *FROM image_assetNATURAL JOIN productsLIMIT 15;-- 3. The cross product equivalent to the "natural join" query above.SELECT img.product_id,	image_id, variant,	process_class,	studio_code,	product_lineFROM image_asset 	AS imgJOIN products 		AS prdON img.product_id = prd.product_idLIMIT 15;-- 4. A query involving a “Group by”, perhaps also with a “HAVING”.-- For each studio get the variant name and number of images belonging to that variant, -- if there are between 3 and 6 images of those specific variants from one particular studio.-- Sort them in descending order.SELECT studio_code,	variant,	COUNT(image_id) AS image_count,	CASE WHEN COUNT(image_id) = 3 THEN 'three'		WHEN COUNT(image_id) = 4 THEN 'four'		WHEN COUNT(image_id) = 5 THEN 'five'		ELSE 'six' END		AS image_count_in_lettersFROM image_assetWHERE variant LIKE 'M%'	OR variant LIKE 'FR%'	OR variant LIKE '%9'GROUP BY studio_code, variantHAVING COUNT(image_id) >= 3 AND COUNT(image_id) < 7ORDER BY image_count DESC;-- 5. A query which uses a sub query.-- Get the unique products along with their category, provided that unique product has 6 or more images.-- Sort by product_id.SELECT product_id, product_lineFROM productsWHERE product_id IN (SELECT product_id	  					FROM image_asset						GROUP BY product_id						HAVING COUNT(image_id) > 5) ORDER BY product_line;-- A query satisfying requirements both 4 and 5.-- This is a performance report.-- Report the number of images, points achieved and average point per image achieved by an associate -- along with the detail of whom they report to and if they are present/absent.-- The report is only for associates who report to a lead who has more than 4 reportees and provided-- that associates have performed some quantifiable work.SELECT n.user_id, t.reports_to,	COUNT(image_id),	SUM(points) 			AS points_achieved,	ROUND(AVG(points), 2) 	AS average_points, 	is_activeFROM n_points 				AS n	JOIN points_table 		AS p		ON n.workflow_event = p.workflow_event	JOIN team 				AS t		ON n.user_id = t.user_idWHERE n.user_id IN (SELECT user_id				  		FROM team				  		WHERE reports_to LIKE ANY (SELECT reports_to							  							FROM team							  							GROUP BY reports_to							  							HAVING COUNT(*) > 4))GROUP BY n.user_id, is_active, t.reports_toORDER BY points_achieved DESC;-- 6. A cross product which cannot be implemented using the words “natural join” (e.g. self join) -- Get the number of reportee for every manager/lead. SELECT t2.reports_to,	COUNT(t1.user_id)FROM team AS t1JOIn team AS t2USING (user_id)GROUP BY t2.reports_to;
